Q3."Write the promise function to print numbers from 1 to 7, in which 1 should be printed after 1 sec, 2 should be printed after 2 sec, 3 should be printed after 3 sec, and so on. Numbers
1
2
3

5

7"
      function printNumber(num) {
        return new Promise(resolve => {
          setTimeout(() => {
            console.log(num);
            resolve();
          }, num * 1000);
        });
      }
      
      async function printNumbers() {
        for (let i = 1; i <= 7; i++) {
          await printNumber(i);
        }
      }
      
      printNumbers();
Q4."Create a promise function accepting an argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed then it should go to reject the state and catch the error and print Promise Rejected "
        function checkValue(value) {
            return new Promise((resolve, reject) => {
              if (value === 'Akash') {
                resolve('Promise Resolved');
              } else {
                reject('Promise Rejected');
              }
            });
          }
          
          checkValue('Akash')
            .then((result) => {
              console.log(result);
            })
            .catch((error) => {
              console.log(error);
            });
